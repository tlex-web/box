fn rgb_at(phase: f64) -> (u8, u8, u8) {
    let f = 0.1;
    let r = ((f * phase + 0.0).sin() * 127.0 + 128.0) as u8;
    let g = ((f * phase + 2.0 * std::f64::consts::PI / 3.0).sin() * 127.0 + 128.0) as u8;
    let b = ((f * phase + 4.0 * std::f64::consts::PI / 3.0).sin() * 127.0 + 128.0) as u8;
    (r, g, b)
}
